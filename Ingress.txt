apiVersion: networking.k8s.io/v1beta1
kind: Ingress
metadata:
  name: basic-routing
spec: 
  rules:                                     #we create rules for all the host we  want
  - host: domain.com
    http:
      paths:
      - path: /
        backend:
          serviceName: fleetman-webapp       #you can check to which service you wanna map
          servicePort: 80

  - host: 192.168.1.1
    http:
      paths:
      - path: /
        backend:
          serviceName: fleetman-queue         #you can check to which service you wanna map
          servicePort: 8161

  - host: www.domain.com
